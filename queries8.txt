-- 1. Create Database and use it
CREATE DATABASE IF NOT EXISTS chit19;
USE chit19;

-- 2. Create Employee Table
CREATE TABLE emp (
    Eno INT PRIMARY KEY AUTO_INCREMENT,      -- Primary Key with auto increment
    Ename VARCHAR(50) NOT NULL,              -- Not Null constraint
    Address VARCHAR(100) DEFAULT 'Nashik',   -- Default value
    Joindate DATE,
    Salary DECIMAL(10,2)
);

-- Set starting AUTO_INCREMENT to 101
ALTER TABLE emp AUTO_INCREMENT = 101;

-- 3. Add Post column
ALTER TABLE emp
